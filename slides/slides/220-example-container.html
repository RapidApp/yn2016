Example rapi/psgi container setup:

> docker create \
    --name=logview --hostname=logview \
    --interactive --tty \
    -p 5000:5000 \
    -v $(pwd)/Rapi-Demo-LogView:/opt/app \
  rapi/psgi
    
> docker start logview
    
> docker exec -it logview bash

Bootstrap via docker:

> mkdir NewApp/

> docker create \
    --name=newapp --hostname=newapp \
    --interactive --tty \
    -p 5000:5000 \
    -v $(pwd)/NewApp:/opt/app \
  rapi/psgi
    
> docker start newapp
    
> docker exec -it newapp bash
